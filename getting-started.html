This is the quick&dirty way to start a D3 project: start creating a svg centered with the given margins 

var margin = {top: 20, right: 20, bottom: 30, left: 40},
    width = 960 - margin.left - margin.right,
    height = 500 - margin.top - margin.bottom;

var svg = d3.select("body").append("svg")
    .attr("width", width + margin.left + margin.right)
    .attr("height", height + margin.top + margin.bottom)
  .append("g")
    .attr("transform", "translate(" + margin.left + "," + margin.top + ")");

After that probably you want to define some scales that maps the values of your data (the domain) into the coordinates (the range) of the graph; if you want something linear (note that the range part of the y axis is reversed) below the example: 

var x = d3.scale.ordinal()
    .rangeRoundBands([0, width], .1)
    .domain([0, 1])
    ;
var y = d3.scale.linear()
    .range([height, 0])
    .domain([0, 1])
    ;

Obviously you can associate an axes to the scale: 

var xAxis = d3.svg.axis()
    .scale(x)
    .orient("bottom");

var svgXAxis = svg.append("g")
      .attr("class", "x axis")
      .attr("transform", "translate(0," + height + ")")
      .call(xAxis);

var color = d3.scale.category10();
